Arduino Code

        int pin1 = 13; // blue wire
        int pin2 = 12; // white wire
        int pin3 = 11;// orange wire

        // the setup function runs once when you press reset or power the board
        void setup() {
          // initializes the 3 pins as an output.
          pinMode(pin1, OUTPUT);
          pinMode(pin2, OUTPUT);
          pinMode(pin3, OUTPUT);
        }

        // the loop function runs over and over again forever
        void loop() {
          // turns pin1 on, keeps the other 2 off
          digitalWrite(pin1, HIGH);
          digitalWrite(pin2, LOW); // turns pin2 off
          digitalWrite(pin3, LOW); // turns pin3 off
          // adds a 1 second delay
          delay(1000);
          // turns pin2 on, keeps the other 2 off
          digitalWrite(pin2, HIGH);
          digitalWrite(pin1, LOW); // turns pin1 off
          digitalWrite(pin3, LOW); // turns pin3 off
          // adds a 1 second delay
          delay(1000);
          // turns pin3 on, keeps other 2 off
          digitalWrite(pin3, HIGH);
          digitalWrite(pin1, LOW); // turns pin1 off
          digitalWrite(pin2, LOW); // turns pin2 off
          delay(1000);
        }
